✅ Vérifier une liste de KBs requis

          
Compare les KBs installés contre une liste obligatoire, rapport coloré

        
      

      
# ── Adapter la liste selon le Patch Tuesday du mois ──
$KBsRequis = @(
    "KB5034763",  # Windows 10 22H2
    "KB5034766",  # Windows 11 23H2
    "KB5034765",  # Windows Server 2022
    "KB5034768"   # Windows Server 2019
)

Write-Host "=== Vérification KBs requis ===" -ForegroundColor Cyan
Write-Host "Machine : $env:COMPUTERNAME  |  $(Get-Date -Format 'yyyy-MM-dd HH:mm')" -ForegroundColor Gray

$res = foreach ($kb in $KBsRequis) {
    $hf = Get-HotFix -Id $kb -ErrorAction SilentlyContinue
    [PSCustomObject]@{
        KB     = $kb
        Statut = if ($hf) { "✅ INSTALLÉ" } else { "❌ MANQUANT" }
        Date   = if ($hf) { $hf.InstalledOn } else { "—" }
        Par    = if ($hf) { $hf.InstalledBy } else { "—" }
    }
}

$res | Format-Table -AutoSize

$manquants = $res | Where-Object Statut -like "*MANQUANT*"
if ($manquants) {
    Write-Host "⚠️  $($manquants.Count) KB(s) manquant(s) — patching requis !" -ForegroundColor Red
} else {
    Write-Host "✅ Tous les KBs sont à jour." -ForegroundColor Green
}

🔄 Mises à jour Windows en attente

          
Détecte les updates non installées via Windows Update Agent (pas de module tiers requis)

        
      

      
# Vérification via COM Windows Update Agent — fonctionne sans PSWindowsUpdate
function Get-PendingWU {
    param([string]$Computer = $env:COMPUTERNAME)

    $sb = {
        $session  = New-Object -ComObject Microsoft.Update.Session
        $searcher = $session.CreateUpdateSearcher()
        try {
            $result = $searcher.Search("IsInstalled=0 AND Type='Software' AND IsHidden=0")
            $result.Updates | Select-Object Title,
                @{N='KB';E={ ($_.KBArticleIDs | ForEach-Object {"KB$_"}) -join ', ' }},
                @{N='Severity';E={ $_.MsrcSeverity }},
                @{N='Size_MB';E={ [math]::Round($_.MaxDownloadSize/1MB,1) }}
        } catch {
            Write-Warning "WUA inaccessible: $_"
        }
    }

    if ($Computer -eq $env:COMPUTERNAME) {
        $updates = & $sb
    } else {
        $updates = Invoke-Command -ComputerName $Computer -ScriptBlock $sb -ErrorAction Stop
    }

    if (!$updates) {
        Write-Host "✅ $Computer — aucune mise à jour en attente" -ForegroundColor Green
    } else {
        Write-Host "⚠️  $Computer — $($updates.Count) update(s) en attente :" -ForegroundColor Yellow
        $updates | Format-Table Title, KB, Severity, Size_MB -AutoSize
    }
}

# Usage :
Get-PendingWU                          # Local
# Get-PendingWU -Computer "SERVEUR01"  # Distant

    


    

      

        
🏢 Audit MSP multi-clients — rapport CSV complet

          
Vérifie un set de KBs critiques sur toutes les machines de plusieurs clients, exporte un rapport

        
      

      
# ══════════════════════════════════════════════════════════════
#  MSP Patch Audit — multi-clients
#  Modifier $KBs et $Clients selon vos environnements
# ══════════════════════════════════════════════════════════════

$KBs = @(
    "KB5034763",   # Win 10 22H2
    "KB5034766",   # Win 11 23H2
    "KB5034765",   # WinSrv 2022
    "KB5034768"    # WinSrv 2019
)

$Clients = @(
    @{ Nom="Client-A"; PCs=@("CA-PC01","CA-PC02","CA-SRV01") },
    @{ Nom="Client-B"; PCs=@("CB-WS01","CB-WS02","CB-DC01")  }
    # ── Ajouter d'autres clients ──
)

$Results  = [System.Collections.Generic.List[PSCustomObject]]::new()
$DateAudit = Get-Date -Format "yyyy-MM-dd HH:mm"

foreach ($client in $Clients) {
    Write-Host "`n═══ Client : $($client.Nom) ═══" -ForegroundColor Cyan

    foreach ($pc in $client.PCs) {
        Write-Host "  → $pc" -NoNewline

        try {
            $hf = Invoke-Command -ComputerName $pc -ErrorAction Stop -ScriptBlock {
                Get-HotFix | Select-Object HotFixID, InstalledOn
            }
            $os = Invoke-Command -ComputerName $pc -ScriptBlock {
                (Get-CimInstance Win32_OperatingSystem).Caption
            }

            foreach ($kb in $KBs) {
                $trouve = $hf | Where-Object HotFixID -eq $kb
                $Results.Add([PSCustomObject]@{
                    Date        = $DateAudit
                    Client      = $client.Nom
                    Machine     = $pc
                    OS          = $os
                    KB          = $kb
                    Statut      = if ($trouve) { "INSTALLE" } else { "MANQUANT" }
                    DateInstall = if ($trouve) { $trouve.InstalledOn } else { "" }
                    Erreur      = ""
                })
            }
            Write-Host " ✅" -ForegroundColor Green

        } catch {
            Write-Host " ❌  ($($_.Exception.Message))" -ForegroundColor Red
            foreach ($kb in $KBs) {
                $Results.Add([PSCustomObject]@{
                    Date="$DateAudit"; Client=$client.Nom; Machine=$pc; OS="INACCESSIBLE"
                    KB=$kb; Statut="INACCESSIBLE"; DateInstall=""; Erreur=$_.Exception.Message
                })
            }
        }
    }
}

# ── Résumé ──
$manquants = $Results | Where-Object Statut -eq "MANQUANT"
Write-Host "`n════ RÉSUMÉ ════" -ForegroundColor Cyan
Write-Host "Vérifications : $($Results.Count)"
Write-Host "Installés     : $(($Results|?{$_.Statut -eq 'INSTALLE'}).Count)" -ForegroundColor Green
Write-Host "Manquants     : $($manquants.Count)"  -ForegroundColor $(if($manquants.Count){"Red"}else{"Green"})
Write-Host "Inaccessibles : $(($Results|?{$_.Statut -eq 'INACCESSIBLE'}).Count)" -ForegroundColor Yellow

if ($manquants) {
    Write-Host "`nMachines prioritaires :" -ForegroundColor Red
    $manquants | Group-Object Machine |
        ForEach-Object { Write-Host "  $($_.Name) : $($_.Group.KB -join ', ')" -ForegroundColor Red }
}

# ── Export CSV ──
$csv = "C:\Temp\MSP_PatchAudit_$(Get-Date -Format 'yyyyMMdd_HHmm').csv"
$Results | Export-Csv $csv -NoTypeInformation -Encoding UTF8
Write-Host "`n📄 Rapport : $csv" -ForegroundColor Cyan

    


    

      

        
📡 Forcer Windows Update + redémarrage si nécessaire

          
Déclenche un scan WU, télécharge et installe les mises à jour critiques (requiert PSWindowsUpdate)

        
      

      
# Nécessite le module PSWindowsUpdate (à installer une fois) :
# Install-Module PSWindowsUpdate -Force -Scope CurrentUser

# ── Installation sur une machine distante ──
param(
    [string[]]$Computers      = @($env:COMPUTERNAME),
    [switch]  $AutoReboot     = $false,
    [string]  $SeverityFilter = "Critical,Important"   # ou "" pour tout installer
)

foreach ($pc in $Computers) {
    Write-Host "`n[$pc] Lancement Windows Update…" -ForegroundColor Cyan

    $params = @{
        ComputerName = $pc
        AcceptAll    = $true
        Install      = $true
        Verbose      = $false
    }
    if ($SeverityFilter) {
        $sevs = $SeverityFilter -split ','
        $params.MicrosoftUpdate = $true
    }
    if ($AutoReboot) { $params.AutoReboot = $true }

    try {
        $updates = Get-WindowsUpdate @params
        if (!$updates) {
            Write-Host "  ✅ Aucune mise à jour à appliquer" -ForegroundColor Green
        } else {
            Write-Host "  ✅ $($updates.Count) mise(s) à jour installée(s)" -ForegroundColor Green
            $updates | Select-Object KB, Title, Status | Format-Table -AutoSize
        }
    } catch {
        Write-Host "  ❌ Erreur : $_" -ForegroundColor Red
    }
}
